\def\@makechapterhead#1{%
	\vspace*{50\p@}%
	{\parindent \z@ \raggedleft
		\ifnum \c@secnumdepth >\m@ne
		\printchapternum
		\par\nobreak
		\vskip 20\p@
		\fi
		\interlinepenalty\@M
		\printchaptertitle{#1}\par\nobreak
		\vskip 40\p@
}}
